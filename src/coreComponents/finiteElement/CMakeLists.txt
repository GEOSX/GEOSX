  set( dependencyList ${dependencyList} cuda )
endif()

                 SOURCES               ${finiteElement_sources}
                 HEADERS               ${finiteElement_headers}

                 OBJECT                ${buildAsObj}
               )

                                   --include ${CMAKE_BINARY_DIR}/include
